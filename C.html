void average3()
{
int numi,c;
float d,k,average_x=0,average_y=0,aver2_x=0,average_xy=0,x[101],y[101];
printf("\n现在进行的是最小二乘法处理数据\n");

scanf("%d",&numi);
for(c=1;c<=numi;c++)
{
printf("请输入第%d个自变量X%d的值",c,c);
scanf("%f",&x[c]);
}
for(c=1;c<=numi;c++)
{
printf("请输入第%d个应变量的Y%d的值",c,c);
scanf("%f",&y[c]);
}
for(c=1;c<=numi;c++)
average_x=average_x+x[c];
average_x=average_x/numi;
for(c=1;c<=numi;c++)
average_y=average_y+y[c];
average_y=average_y/numi;
for(c=1;c<=numi;c++)
aver2_x=aver2_x+x[c]*x[c];
aver2_x=aver2_x/numi;
for(c=1;c<=numi;c++)
average_xy=average_xy+x[c]*y[c];
average_xy=average_xy/numi;
k=(average_x*average_y-average_xy)/(average_x*average_x-aver2_x);
d=average_y-k*average_x;
printf("X的平均值=%f\n",average_x);
printf("Y的平均值=%f\n",average_y);
printf("X*Y的平均值=%f\n",average_xy);
printf("X的平方的平均值=%f\n",aver2_x);
printf("自变量X的值为");
for(c=1;c<=numi;c++)
printf("%f",x[c]);
printf("\n应变量Y的值为");
for(c=1;c<=numi;c++)
printf("%f",y[c]);
printf("\n计算结果：斜率k=%f\n",k);
printf("\n计算结果：截距b=%f\n",d);
}
